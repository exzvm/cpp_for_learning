# 设置cmake版本号
cmake_minimum_required(VERSION 3.26)

# 设置工程名称
project(cpp_for_learning VERSION 1.0)

# 设置工程包含当前目录
set(CMAKE_INCLUDE_CURRENT_DIR ON)

# 设置CPP标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 添加头文件搜索目录
include_directories(
    ${CMAKE_SOURCE_DIR}/src/hash/include
    ${CMAKE_SOURCE_DIR}/src/hash
    # ${CMAKE_CURRENT_BINARY_DIR}  
)

# 匹配文件
file(GLOB HEADER_FILE ${PROJECT_SOURCE_DIR}/src/*.h)
file(GLOB SOURCE_FILE ${PROJECT_SOURCE_DIR}/src/*.cpp)

# 链接可执行文件
add_executable(${PROJECT_NAME} ${HEADER_FILE} ${SOURCE_FILE})
